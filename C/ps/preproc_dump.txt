PS* load_ps(const char* file_str)
{

 if(!file_str)return ((void *)0);
 int i,n_elements;
 double base_mva;
 FILE *file;
 char curr_line[256];
 PS *ps;
 BUS *buses;
 BRANCH *branches;
 GEN *gens;
 SHUNT *shunts;



 file=fopen(file_str,"r");
 if(!file) return ((void *)0);


 ps=malloc(sizeof(PS));


 if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
 sscanf(curr_line,"BASE_MVA %lg",&base_mva);
 ps->base_mva=base_mva;


 if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
 sscanf(curr_line,"BUS %d",&n_elements);
 ps->nbus=n_elements;
 buses=ps->buses=malloc(n_elements*sizeof(BUS));
 for(i=0;i<n_elements;i++){
  if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
  SCAN_BUS(curr_line,buses+i);
 }





 qsort(ps->buses,ps->nbus,sizeof(BUS),bus_cmp);


 if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
 sscanf(curr_line,"BRANCH %d",&n_elements);
 ps->nbranch=n_elements;
 branches=ps->branches=malloc(n_elements*sizeof(BRANCH));
 for(i=0;i<n_elements;i++){
  if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
  SCAN_BRANCH(curr_line,branches+i);
 }




 for(i=0;i<ps->nbranch;i++){
  branches[i].to=get_bus_idx(ps,branches[i].to);
  branches[i].from=get_bus_idx(ps,branches[i].from);
  if(branches[i].to<branches[i].from){branches[i].to=branches[i].to+branches[i].from;branches[i].from=branches[i].to-branches[i].from;branches[i].to=branches[i].to-branches[i].from;}
 }





 qsort(ps->branches,ps->nbranch,sizeof(BRANCH),branch_cmp);


 if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
 sscanf(curr_line,"GEN %d",&n_elements);
 ps->ngen=n_elements;
 gens=ps->gens=malloc(n_elements*sizeof(GEN));
 for(i=0;i<n_elements;i++){
  if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
  SCAN_GEN(curr_line,gens+i);
 }


 if(!fgets(curr_line,256,file)){ps_free(ps);return ((void *)0);}
 sscanf(curr_line,"SHUNT %d",&n_elements);
 ps->nshunt=n_elements;
 shunts=ps->shunts=malloc(n_elements*sizeof(SHUNT));

 fclose(file);
 return ps;

int get_bus_idx(PS *ps,int id)
{

 if(!ps)return -1;
 if(!ps->buses)return -1;
 int nbus=ps->nbus;
 BUS *buses=ps->buses,*match=((void *)0),*key=malloc(sizeof(BUS));
 key->id=id;


 match=(BUS*)bsearch(key,buses,nbus,sizeof(BUS),bus_cmp);


 free(key);
 if(!match)return -1;
 return match-buses;
}




int* find_subgraphs(PS *ps)
{

 if(!ps)return ((void *)0);
 if(!ps->branches)return ((void *)0);
 int i,j,*r,*p,nbus=ps->nbus,nbranch=ps->nbranch,curr_cmpt=0,*prnt,*child,*wkspc=malloc((nbus)*sizeof(int)),*cmpts=malloc((nbus+1)*sizeof(int));
 prnt=child=wkspc;
 BRANCH *branches=ps->branches;
 cs_di *graph_tmp=cs_di_spalloc(0,0,0,0,1),*graph=((void *)0);


 for(i=0;i<nbus;i++)cmpts[i]=-1;


 for(i=0;i<nbranch;i++){
  if(branches[i].status){
   cs_di_entry(graph_tmp,branches[i].from,branches[i].to,0.0);
   cs_di_entry(graph_tmp,branches[i].to,branches[i].from,0.0);
  }
 }
 graph=cs_di_compress(graph_tmp);
 cs_di_spfree(graph_tmp);
 graph_tmp=((void *)0);
 p=graph->p;
 r=graph->i;
# 228 "ps.c"
 for(i=0;i<nbus;i++){
  if(cmpts[i]!=-1)continue;
  cmpts[i]=curr_cmpt;
  *prnt=i;
  do{
   for(j=p[*prnt];j<p[*prnt+1];j++){
    if(cmpts[r[j]]==-1){
     cmpts[r[j]]=curr_cmpt;
     *(++child)=r[j];
    }
   }
  }while(prnt++<child);
  prnt--;
  curr_cmpt++;
 }


 free(wkspc);
 cs_di_spfree(graph);
 cmpts[nbus]=curr_cmpt;
 return cmpts;
}
